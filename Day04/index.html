// Gera um número aleatório entre 0 e 10
let numeroSecreto = Math.floor(Math.random() * 11);
let tentativas = 3;
let acertou = false;

alert("Bem-vindo ao jogo de adivinhação! Você tem 3 tentativas para acertar o número entre 0 e 10.");

while (tentativas > 0) {
    let chute = parseInt(prompt("Tente adivinhar o número entre 0 e 10:"));
    
    if (chute === numeroSecreto) {
        alert("Parabéns! Você acertou!");
        acertou = true;
        break; // Sai do loop se acertar
    } else {
        tentativas--;
        if (tentativas > 0) {
            let dica = chute < numeroSecreto ? "O número é maior!" : "O número é menor!";
            alert(`Errado! ${dica} Você tem mais ${tentativas} tentativa(s).`);
        } else {
            alert(`Suas tentativas acabaram! O número era ${numeroSecreto}.`);
        }
    }
}

if (!acertou) {
    alert("Que pena! Tente novamente em outra rodada.");
}

// Pergunta se o jogador quer jogar novamente
let jogarNovamente = prompt("Quer jogar de novo? (s/n)").toLowerCase();

while (jogarNovamente === "s") {
    numeroSecreto = Math.floor(Math.random() * 11);
    tentativas = 3;
    acertou = false;

    }

}

alert("Obrigado por jogar! Até a próxima.");
