7DaysOfCode - Day 05

Código

// Criamos arrays vazios para armazenar os itens de cada categoria
let frutas = [];
let laticinios = [];
let congelados = [];
let doces = [];
let outros = [];

// Variável para controlar a repetição do loop
let adicionarMais = true;

// Iniciamos o loop enquanto o usuário quiser adicionar itens
while (adicionarMais) {
    let resposta = prompt("Você deseja adicionar uma comida à sua lista de compras? (sim/não)").toLowerCase();
    
    if (resposta === "não") {
        adicionarMais = false; // Sai do loop
    } else if (resposta === "sim") {
        let comida = prompt("Qual comida você deseja adicionar?");
        let categoria = prompt("Em qual categoria essa comida se encaixa? (frutas, laticínios, congelados, doces, outros)").toLowerCase();
        
        // Adiciona o item na categoria correta
        if (categoria === "frutas") {
            frutas.push(comida);
        } else if (categoria === "laticínios") {
            laticinios.push(comida);
        } else if (categoria === "congelados") {
            congelados.push(comida);
        } else if (categoria === "doces") {
            doces.push(comida);
        } else {
            outros.push(comida);
        }
    } else {
        alert("Resposta inválida! Digite 'sim' ou 'não'.");
    }
}

// Exibir a lista de compras formatada
console.log("Lista de compras:");

if (frutas.length > 0) {
    console.log(`Frutas: ${frutas.join(", ")}`);
}
if (laticinios.length > 0) {
    console.log(`Laticínios: ${laticinios.join(", ")}`);
}
if (congelados.length > 0) {
    console.log(`Congelados: ${congelados.join(", ")}`);
}
if (doces.length > 0) {
    console.log(`Doces: ${doces.join(", ")}`);
}
if (outros.length > 0) {
    console.log(`Outros: ${outros.join(", ")}`);
}
